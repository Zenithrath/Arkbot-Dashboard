## 5. Validasi File Upload

**File:** `src/pages/UploadPage.tsx`

**Validasi format file:**
- Definisikan allowed types: `.pdf`, `.md` (sekarang), `.xlsx`, `.xls`, `.docx` (nanti)
- Validasi di `addFiles` function sebelum menambahkan ke items
- Tampilkan toast error untuk format yang tidak didukung

**Validasi file kosong:**
- Cek `file.size === 0` di `addFiles` function
- Tampilkan toast error untuk file kosong

**Implementasi:**
```typescript
const ALLOWED_EXTENSIONS = ['.pdf', '.md', '.xlsx', '.xls', '.docx']

const addFiles = (files: FileList | null) => {
  if (!files) return
  const newItems: UploadItem[] = []
  
  Array.from(files).forEach((file) => {
    const ext = '.' + file.name.split('.').pop()?.toLowerCase()
    if (!ALLOWED_EXTENSIONS.includes(ext)) {
      toast.error(`Format ${ext} tidak didukung: ${file.name}`)
      return
    }
    if (file.size === 0) {
      toast.error(`File kosong: ${file.name}`)
      return
    }
    newItems.push({ id: uuidv4(), file, status: "pending" })
  })
  
  setItems((prev) => [...prev, ...newItems])
}
```

try {
  const res = await fetch(UPLOAD_URL, {
    method: "POST",
    body: formData,
    signal: AbortSignal.timeout(30000), // 30 detik timeout
  })

  if (!res.ok) {
    let errorMsg = `HTTP ${res.status}`
    try {
      const errorBody = await res.json()
      if (errorBody.message || errorBody.error) {
        errorMsg = errorBody.message || errorBody.error
      }
    } catch {}
    throw new Error(errorMsg)
  }

  const data = await res.json().catch(() => ({}))
  if (data.status === "error" || data.error) {
    throw new Error(data.message || data.error || "Workflow error")
  }

  // success...
} catch (err) {
  let message = "Upload failed"
  if (err instanceof TypeError && err.message.includes("fetch")) {
    message = "Server tidak dapat dijangkau. Periksa koneksi internet."
  } else if (err instanceof DOMException && err.name === "TimeoutError") {
    message = "Upload timeout. Server tidak merespon."
  } else if (err instanceof Error) {
    message = err.message
  }
  // update status...
}
